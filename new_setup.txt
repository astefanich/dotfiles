ln -sf $HOME/dotfiles/shells/zsh/.zprofile $HOME/.zprofile

ln -sf $HOME/dotfiles/editors/vim/.vim $HOME/.vim
ln -sf $HOME/dotfiles/editors/vim/.vimrc $HOME/.vimrc
