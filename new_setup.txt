git clone https://github.com/veyga/dotfiles

Install:
zsh
tmux
nvim
ripgrep?
sdkman?
npm? (required for CoC)
nerdfonts? (only needed for devicons)

chsh -s $(which zsh)

(enter nvim)
:PlugInstall everything

ln -sf $HOME/dotfiles/shells/zsh/.zshenv $HOME/.zshenv
ln -sf $HOME/dotfiles/shells/zsh/.zprofile $HOME/.zprofile
ln -sf $HOME/dotfiles/editors/nvim $HOME/config/nvim
sudo ln -sf $HOME/dotfiles/editors/nvim/theme/obsidian.vim /usr/share/nvim/runtime/colors/obsidian.vim
